# General

program -> declarationList
declarationList -> declarationList declaration | declaration
declaration -> varDec | functionDeclaration

# Variable Declarations

varDec -> typeSpecifier ID = expression ;
# varDec -> typeSpecifier varDecList ;
# varDeclist -> varDecList, varDecInit | varDecInit
# varDecInit -> ID | ID = expression

# Function Declarations

functionDeclaration -> typeSpecifier ID ( ) { statementList }

# Statements

statementList -> statementList statement | statement
statement -> varDec | returnStatement
returnStatement -> return expression ;

# Expressions

expression -> t + expression | t - expression | t
t -> e * t | e / t | e
e -> constNum | ID
