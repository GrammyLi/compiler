program -> statementList
statementList -> statementList statement | statement
statement -> functionStatement | returnStatement | ifStatement
functionStatement -> typeSpecifier ID ( ) { statementList }
returnStatement -> return expression ;
expression -> t | expression + t | expression - t
t -> e | t * e | t / e
e -> constNum | ID
assignment -> typeSpecifier ID = expression ;
ifStatement -> if ( condition ) { statementList }
condition -> ID == constNum
