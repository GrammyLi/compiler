program -> statementList
statementList -> statementList statement | statement
statement -> functionStatement | returnStatement | assignmentStatement | ifStatement | forStatement
functionStatement -> typeSpecifier ID ( ) { statementList }
returnStatement -> return constNum ; | return expression ;
expression -> s
s -> t | s + t | s - t
t -> e | t * e | t / e
e -> p | p ^ e
p -> ID | constNum | ( s )
ifStatement -> if ( condition ) { statementList }
condition -> ID == constNum
assignmentStatement -> typeSpecifier ID = constNum ; | ID = constNum ; | ID += constNum ;
forStatement -> for ( ID = constNum ; ID < ID ; ID ++ ) { statementList }
